#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int n, k;
        string s;
        cin >> n >> k >> s;

        vector<int> parity(k, 0);

        for (int i = 0; i < n; ++i) {
            if (s[i] == '1') {
                parity[i % k] ^= 1;
            }
        }

        bool possible = true;
        for (int r = 0; r < k; ++r) {
            if (parity[r]) {
                possible = false;
                break;
            }
        }

        cout << (possible ? "YES\n" : "NO\n");
    }
}
