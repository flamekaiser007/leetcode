#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        long long n;
        cin >> n;

        if (n == 10) {
            cout << -1 << '\n';
            continue;
        }

        long long a = n % 12;

        if (a == 10) {
            a = 22;
        }

        cout << a << ' ' << n - a << '\n';
    }
}
